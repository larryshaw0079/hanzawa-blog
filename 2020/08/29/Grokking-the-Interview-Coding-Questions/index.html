  • 快速排序
// 要修改原数组的内容，引用&是必须的
void QuickSort(std::vector<int> &arr) {
    std::vector<std::pair<int, int>> ranges; // 用来保存需要迭代的范围
    ranges.push_back(std::make_pair(0, arr.size()-1)); // 初始化，闭区间
    
    while (!ranges.empty()) {
        auto range = ranges.back(); // 取出栈顶
        ranges.pop_back();
        
        if (range.first >= range.second) continue; // 只有一个数的情况直接跳过
        
        int mid_val = arr[range.second]; // 最后一个数作为anchor
        int left = range.first, right = range.second-1;
        
        while (left < right) {
            while (left < right && arr[left] < mid_val) left++;
            while (left < right && arr[right] >= mid_val) right--;
            std::swap(arr[left], arr[right]);
        }
        
        // 处理特殊情况
        if (arr[left] >= arr[range.second]) std::swap(arr[left], arr[range.second]);
        else left++;
        
        ranges.push_back(std::make_pair(range.first, left-1));
        ranges.push_back(std::make_pair(left+1, range.second));
    }
}

Stock Series 股票系列问题

买卖股票的最佳时机

🐱‍💻Leetcode 121 Easy

暴力的解法就不说了，这里给的是一次遍历的做法，假设当前是第\(K\)天，如果要在当天卖出，那么最后价格肯定是第\(K\)天的价格减去前\(K-1\)天的最低价，同时前\(K-1\)天的最低价是可以边遍历边更新的，所以最终只需要一次遍历即可：

class Solution {
 public:
  int maxProfit(std::vector<int>& prices) {
    int min_price = INT_MAX, max_profit = 0;
    for (auto price : prices) {
      max_profit = max_profit > price - min_price ? max_profit : price - min_price;
      // min_price一定要后更新，因为min_price代表的是当前天以前的最低价格
      min_price = price < min_price ? price : min_price;
    }
    return max_profit;
  }
};

Top K Element 第K大/小元素

🐱‍💻Leetcode 215 Medium （同类题：🐱‍💻剑指Offer 40 Easy）

题目中说了给定的\(K\)是合法的，所以下面的代码没有检查\(K\)的合法性

  • 先排序再求解 \(O(N\log N)\) （⭐）
class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        std::sort(nums.begin(), nums.end());
        return nums[nums.size()-k];
    }
};
  • 堆 \(O(N\log K)\) （⭐⭐）

注意求最小的\(K\)个数的话用大顶堆，不然把全部数push到堆中然后一个一个弹出的话复杂度是\(O(N\log N)\)，和排序没什么区别了。

使用C++自带priority_queue的版本：

int TopK(std::vector<int>& arr, int k) {
    std::priority_queue<int, std::vector<int>, std::greater<int>> pq; // greater是小顶堆，less是大顶堆
    for (auto num : arr) {
        if (pq.size() < k) {
            pq.push(num);
        } else if (num > pq.top()) {
            // 如果栈顶元素大于当前元素
            pq.pop();
            pq.push(num);
        }
    }
    
    return pq.top();
}

class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        return TopK(nums, k);
    }
};
  • 快排变体 \(O(N)\) （⭐⭐⭐）
int TopK(std::vector<int>& arr, int k) {
  std::vector<std::pair<int, int>> ranges;
  ranges.push_back(std::make_pair(0, arr.size()-1));

  while (!ranges.empty()) {
    auto range = ranges.back();
    ranges.pop_back();

    if (range.first >= range.second) {
      if (range.first == k || range.second == k) {
        return arr[k];
      }
      continue;
    }
    int mid_val = arr[range.second];
    int left = range.first, right = range.second-1;
    while (left < right) {
      while (left < right && arr[left] < mid_val) left++;
      while (left < right && arr[right] >= mid_val) right--;
      std::swap(arr[left], arr[right]);
    }

    if (arr[left] >= arr[range.second]) std::swap(arr[left], arr[range.second]);
    else left++;

    if (left == k) return arr[left];

    ranges.push_back(std::make_pair(range.first, left-1));
    ranges.push_back(std::make_pair(left+1, range.second));
  }
  return 0;
}

class Solution {
public:
    int findKthLargest(vector<int>& nums, int k) {
        return TopK(nums, nums.size()-k);
    }
};

Intersection of Two Lists 链表相交节点

🐱‍💻Leetcode 160 Easy

  • HashTable法 \(O(M+N)\)（⭐⭐）

空间复杂度\(O(M)\)或\(O(N)\)，这里使用unordered_set实现：

class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        if (headA == nullptr || headB == nullptr) return nullptr;

        std::unordered_set<ListNode*> us;
        
        ListNode* current = headA;
        while (current) {
            us.insert(current);
            current = current->next;
        }

        ListNode* res = nullptr;

        current = headB;
        while (current) {
            if (us.find(current) != us.end()) {
                res = current;
                break;
            }
            current = current->next;
        }

        return res;
    }
};
  • 双指针法 \(O(M+N)\)（⭐⭐⭐）

空间复杂度\(O(1)\)

class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {
        if (headA == nullptr || headB == nullptr) return nullptr;

        ListNode* a = headA, *b = headB;

        while (a != b) {
            if (a == nullptr) {
                a = headB;
            } else {
                a = a->next;
            }

            if (b == nullptr) {
                b = headA;
            } else {
                b = b->next;
            }
        }

        return a;
    }
};

Reverse the List 翻转链表

🐱‍💻Leetcode 206 Easy

面试中出现频率还挺高的

class Solution {
public:
    ListNode* enumerate(ListNode* head) {
        std::vector<int> buffer;
        while (head != nullptr) {
            buffer.push_back(head->val);
            head = head->next;
        }
        ListNode* res = new ListNode(buffer.back());
        ListNode* current = res;
        for (int i = buffer.size()-2; i >= 0; i--) {
            current->next = new ListNode(buffer[i]);
            current = current->next;
        }

        return res;
    }

    ListNode* reverseList(ListNode* head) {
        if (head == nullptr) return head;

        return enumerate(head);
    }
};

Merge Lists 合并（2或K个）链表

🐱‍💻Leetcode 23 Hard

最简单的hard题🤣？？？优先队列法写起来是最舒服的，效率也是最高的，时间复杂度\(O(nk\times\log nk)\)

class Solution {
    std::priority_queue<std::pair<int, ListNode*>, std::vector<std::pair<int, ListNode*>>, std::greater<std::pair<int, ListNode*>>> q;

public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        if (lists.empty()) return nullptr;

        for (auto list : lists) {
            if (list != nullptr) {
                q.push(std::make_pair(list->val, list));
            }
        }

        auto* dummy = new ListNode(-1);
        auto* current = dummy;

        while (q.size()) {
            auto pair = q.top();
            auto* p = pair.second;
            q.pop();

            current->next = p;
            if (p->next) q.push(std::make_pair(p->next->val, p->next));
            current = current->next;
        }

        return dummy->next;
    }
};

Find Median from Data Stream 找到数据流的中位数

🐱‍💻Leetcode 295 Hard

双堆做法：

class MedianFinder {
public:
    std::priority_queue<int, std::vector<int>, std::less<int>> big_root;
    std::priority_queue<int, std::vector<int>, std::greater<int>> small_root;
    
    /** initialize your data structure here. */
    MedianFinder() {
        
    }
    
    void addNum(int num) {
        big_root.push(num);
        int tmp = big_root.top();
        big_root.pop();
        small_root.push(tmp);

        if (small_root.size() > big_root.size()) {
            tmp = small_root.top();
            small_root.pop();
            big_root.push(tmp);
        }
    }
    
    double findMedian() {
        if (big_root.size() == small_root.size()) {
            return (static_cast<double>(big_root.top()) + small_root.top())/2.0;
        } else {
            return static_cast<double>(big_root.top());
        }
    }
};

Power of Two 判断\(2\)的幂

🐱‍💻Leetcode 231 Easy

\(n\)为\(2\)的幂只需检查\(n\)与\(n-1\)的与是否为\(0\)即可，如\(8\)的二进制表示（假设只有\(8\)位）为[0001000]，\(7\)的二进制为[0000111]。

一行解决。记得判断大于\(0\)。

class Solution {
public:
    bool isPowerOfTwo(int n) {
        return n > 0 && (n&(n-1))==0;
    }
};

class Solution {
